\begin{frame}{Active Learning}
\begin{itemize}
    \onslide<1-> \item \textbf{Goal:} Recognize most relevant examples and query their labels from
        an all-knowing oracle
    \onslide<2-> \item \textbf{Issue:} Traditional active learning works poorly for modern
        neural networks, especially during \textit{cold-start}
    \onslide<3-> \item Limitations in \abr{sota} \nlp{} show a greater need for active learning
        \textit{and} make active learning more difficult to deploy
\end{itemize}
\end{frame}
