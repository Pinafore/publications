\begin{frame}{\nlp{} for Low-resource Languages}
\begin{itemize}
    \onslide<1->\item Scarcity of both labeled and unlabeled data holds back applications in
    low-resource languages
\onslide<2->\item Cross-lingual word embeddings (\abr{clwe}) can bridge the gap by mapping
    words from different languages to a shared vector space
\onslide<3->\item \abr{clwe} not tailored for downstream tasks and need
    refinement~\citep{zhang-20b}
\end{itemize}
\onslide<4->\textbf{How can we quickly refine CLWE for low-resource NLP?}
\end{frame}
